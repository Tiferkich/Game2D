30 if (skillLvl == 0) return -1;
31 if (dungeon.getHero().first->getCharacteristics().getCurMana() < manaCost * skillLvl) return -2;
32 Move side = dungeon.getHero().first->getSideLooking();
33 int y = dungeon.getHero().first->getCoords().first;
34 int x = dungeon.getHero().first->getCoords().second;
35 Enemy *enemy = nullptr;
36 int h = 10000000;
37 auto item = dungeon.getListEnemies().begin();
38 for (item = dungeon.getListEnemies().begin(); item != dungeon.getListEnemies().end(); ++item) {
39 if (abs(item->first->getCoords().second - x) + abs(item->first->getCoords().first - y) < h &&
40 check(side, x, y, item->first->getCoords().second, item->first->getCoords().first) &&
41 item->first->getStatus() != Status::dead &&
42 (item->first->getFaction() == Faction::character || item->first->getFaction() == Faction::undead)) {
43 h = abs(item->first->getCoords().second - x) + abs(item->first->getCoords().first - y);
44 enemy = item->first;
45 }
46 }
47 if (h > skillRange) return -3;
48 if (enemy == nullptr) return -4;
49 if (enemy->getFaction() == Faction::character) {
50 enemy->setCurHp(enemy->getMaxHp() * (skillLvl + 1));
51 enemy->setMaxHp(enemy->getMaxHp() * (skillLvl + 1));
52 enemy->setDamage({enemy->Damage().first * (skillLvl + 1),
53 enemy->Damage().second * (skillLvl + 1)});
54 }
55 if (enemy->getFaction() == Faction::undead) {
56 enemy->setCurHp(enemy->getMaxHp() / (skillLvl + 1));
57 enemy->setMaxHp(enemy->getMaxHp() / (skillLvl + 1));
58 enemy->setDamage({enemy->Damage().first / (skillLvl + 1),

60 }
61 if (enemy->getCurHp() <= 0) {
62 enemy->die();
63 }
64 dungeon.getHero().first->getCharacteristics().mana(-1 * manaCost * skillLvl / 2);
65 return 0;
66 }
