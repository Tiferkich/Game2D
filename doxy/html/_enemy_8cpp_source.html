10void Enemy::attack(Unit &someone) {
11 if (status == Status::dead) throw (std::invalid_argument("Unit is dead"));
12 if (someone.getFaction() == faction) {
13 status = Status::wandering;
14 throw (std::invalid_argument("The same unit"));
15 }
16 bool flag = false;
17 int dx[9] = {1, 0, -1, 0, 1, 0, 1, -1, -1}; // смещения, соответствующие соседям ячейки
18 int dy[9] = {0, 1, 0, -1, 1, 0, -1, -1, 1};
19 for (int i = 0; i < 6; i++) {
20 if (flag) { break; }
21 int x = coords.second + dx[i];
22 int y = coords.first + dy[i];
23 if (x == someone.getCoords().second && y == someone.getCoords().first) flag = true;
24 }
25
26 if (!flag) {
27 status = Status::wandering;
28 throw (std::invalid_argument("Unit too far away"));
29 }
30 srand(time(0));
31 int damage0 = rand() % (damage.second - damage.first + 1) + damage.first;
32 status = Status::aggresion;
33 someone.getDamage(damage0);
34}

44std::vector<std::pair<int, int>> Enemy::Lee(Matrix<Ceil> &matrix, std::pair<int, int> coordTo) {
45 if (coordTo.first == coords.first && coordTo.second == coords.second)
46 throw (std::invalid_argument("The same coords"));
47 int W = matrix.cols(); // ширина рабочего поля
48 int H = matrix.rows(); // высота рабочего поля
49 int WALL = -1; // непроходимая ячейка
50 int ax = coords.second;
51 int ay = coords.first;
52 int bx = coordTo.second;
53 int by = coordTo.first;
54 int BLANK = -2; // свободная непомеченная ячейка
55 int grid[matrix.rows()][matrix.cols()];
56 for (int j = 0; j < matrix.rows(); j++) {
57 for (int i = 0; i < matrix.cols(); i++) {
58 if (matrix[i][j] == Ceil::wall || matrix[i][j] == Ceil::closeDoor) {
59 grid[i][j] = WALL;
60 } else {
61 grid[i][j] = BLANK;
62 }
63 }
64 }
65 int dx[4] = {1, 0, -1, 0}; // смещения, соответствующие соседям ячейки
66 int dy[4] = {0, 1, 0, -1}; // справа, снизу, слева и сверху
67 int d, x, y, k;
68 int len;
69 int px[W * H], py[W * H];
70 bool stop;
71
72 if (grid[ay][ax] == WALL && grid[by][bx] == WALL)
73 throw std::runtime_error("No way"); // ячейка (ax, ay) или (bx, by) - стена
74
75 // распространение волны
76 d = 0;
77 grid[ay][ax] = 0; // стартовая ячейка помечена 0
78 do {
79 stop = true; // предполагаем, что все свободные клетки уже помечены
80 for (y = 0; y < H; ++y)
81 for (x = 0; x < W; ++x)
82 if (grid[y][x] == d) // ячейка (x, y) помечена числом d
83 {
84 for (k = 0; k < 4; ++k) // проходим по всем непомеченным соседям
85 {
86 int iy = y + dy[k], ix = x + dx[k];
87 if (iy >= 0 && iy < H && ix >= 0 && ix < W &&
88 grid[iy][ix] == BLANK) {
89 stop = false; // найдены непомеченные клетки
90 grid[iy][ix] = d + 1; // распространяем волну
91 }
92 }
93 }
94 d++;
95 } while (!stop && grid[by][bx] == BLANK);
96
97 if (grid[by][bx] == BLANK) throw (std::runtime_error("No way")); // путь не найден
98
99 // восстановление пути
100 len = grid[by][bx]; // длина кратчайшего пути из (ax, ay) в (bx, by)
101 x = bx;
102 y = by;
103 d = len;
104 while (d > 0) {
105 px[d] = x;
106 py[d] = y; // записываем ячейку (x, y) в путь
107 d--;
108 for (k = 0; k < 4; ++k) {
109 int iy = y + dy[k], ix = x + dx[k];
110 if (iy >= 0 && iy < H && ix >= 0 && ix < W &&
111 grid[iy][ix] == d) {
112 x = x + dx[k];
113 y = y + dy[k]; // переходим в ячейку, которая на 1 ближе к старту
114 break;
115 }
116 }
117 }
118 px[0] = ax;
119 py[0] = ay;
120 std::vector<std::pair<int, int>> array;
121 for (int i = 0; i < len; i++) {
122 array.emplace_back(px[i], py[i]);
123 }
124 return array;
125}

128Enemy::move(Matrix<Ceil> &matrix, std::pair<int, int> coordTo, std::list<std::pair<Enemy *, Ceil *>> &listEnemies) {
129
130 if (coordTo.first < 0 || coordTo.second < 0 ||
131 coordTo.second >= matrix.cols() ||
132 coordTo.first >= matrix.rows())
133 throw std::runtime_error("invalid coords");
134
135 if ((coordTo.first - coords.first) * (coordTo.first - coords.first) +
136 (coordTo.second - coords.second) * (coordTo.second - coords.second) > vision * vision * vision) {
137 status = Status::wandering;
138 throw std::invalid_argument("Error distance");
139 }
140 std::vector<std::pair<int, int>> ways = Lee(matrix, {coordTo.first, coordTo.second});
141 if (ways.size() == 1 || ways.empty()) {
142 return;
143 }
144 coords.first = ways[1].first;
145 coords.second = ways[1].second;
146 for (auto &item: listEnemies) {
147 if (item.first == this) {
148 item.second = &matrix[coords.second][coords.first];
149 }
150 }
151
152}

155Unit *Enemy::findTarget(Dungeon &dungeon) {
156 std::pair<int, int> nearDist = {INT32_MAX / 10, INT32_MAX / 10};
157 Unit *unit = nullptr;
158 auto enemy = dungeon.getListEnemies().begin();
159 for (enemy = dungeon.getListEnemies().begin(); enemy != dungeon.getListEnemies().end(); ++enemy) {
160 if ((enemy->first->getCoords().first - coords.first) * (enemy->first->getCoords().first - coords.first) +
161 (enemy->first->getCoords().second - coords.second) * (enemy->first->getCoords().second - coords.second) <
162 nearDist.first * nearDist.first + nearDist.second * nearDist.second &&
163 enemy->first->status != Status::dead && enemy->first->faction != faction &&
164 abs(enemy->first->getCoords().second - coords.second) +
165 abs(enemy->first->getCoords().first - coords.first) <=
166 vision) {
167 nearDist.first = (enemy->first->getCoords().first - coords.first);
168 nearDist.second = (enemy->first->getCoords().second - coords.second);
169 unit = enemy->first;
170 }
171 }
172 if ((dungeon.getHero().first->getCoords().first - coords.first) *
173 (dungeon.getHero().first->getCoords().first - coords.first) +
174 (dungeon.getHero().first->getCoords().second - coords.second) *
175 (dungeon.getHero().first->getCoords().second - coords.second) <
176 nearDist.first * nearDist.first + nearDist.second * nearDist.second &&
177 dungeon.getHero().first->getFaction() != faction &&
178 dungeon.getHero().first->getCharacteristics().getCurHp() > 0 &&
179 abs(dungeon.getHero().first->getCoords().second - coords.second) +
180 abs(dungeon.getHero().first->getCoords().first - coords.first) <= vision) {
181 unit = dungeon.getHero().first;
182 }
183
184
185 return unit;
186}

189void Enemy::doSomething(Dungeon &dungeon) {
190 if (status == Status::dead) {
191 return;
192 }
193 Unit *someone = findTarget(dungeon);
194 if (someone == nullptr) {
195 status = Status::wandering;
196 srand(time(0));
197 int i = rand() % 4;
198 int dx[4] = {1, 0, -1, 0}; // смещения, соответствующие соседям ячейки
199 int dy[4] = {0, 1, 0, -1};
200 if (coords.first + dy[i] < 0 || coords.second + dx[i] < 0 ||
201 coords.second + dx[i] >= dungeon.getMapLevel().cols() ||
202 coords.first + dy[i] >= dungeon.getMapLevel().rows())
203 return;
204 try {
205 move(dungeon.getMapLevel(), {coords.first + dy[i], coords.second + dx[i]}, dungeon.getListEnemies());
206 } catch (...) {}
207 return;
208 }
209 try {
210 move(dungeon.getMapLevel(), {someone->getCoords().first, someone->getCoords().second},
211 dungeon.getListEnemies());
212 status = Status::aggresion;
213 } catch (...) {}
214 try {
215 attack(*someone);
216 status = Status::aggresion;
217 } catch (...) {}
218}
