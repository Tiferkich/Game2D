27 bool check(Move side, int ax, int ay, int bx, int by) {
28 if (side == Move::right) {

31 if (ax >= bx) return true;
32 } else if (side == Move::up) {
33 if (ay <= by) return true;
34 } else if (side == Move::down) { if (ay >= by) return true; }

48 int useSkill(Dungeon &dungeon, int mas[]) override {
49 if (skillLvl == 0) return -1;
50 std::string nameMorph = Game::allTypesEnemies[mas[0]];
51 //if (!dungeon.getHero().first->isOpenEnemy(nameMorph)) return -2;
52 if (dungeon.getHero().first->getCharacteristics().getCurMana() < manaCost * skillLvl) return -3;
53 Move side = dungeon.getHero().first->getSideLooking();
54 int y = dungeon.getHero().first->getCoords().first;
55 int x = dungeon.getHero().first->getCoords().second;
56 Enemy *enemy = nullptr;
57 int h = 10000000;
58 auto item = dungeon.getListEnemies().begin();
59 for (item = dungeon.getListEnemies().begin(); item != dungeon.getListEnemies().end(); ++item) {
60 if (check(side, x, y, item->first->getCoords().second, item->first->getCoords().first) &&
61 abs(item->first->getCoords().second - x) + abs(item->first->getCoords().first - y) < h &&
62 (item->first->getFaction() == Faction::character || item->first->getFaction() == Faction::undead)) {
63 try {
64 if (dungeon.getHero().first->isOpenEnemy(dynamic_cast<Undead *>(item->first)->getNameLiving()) &&
65 item->first->getName() != nameMorph) {
66 h = abs(item->first->getCoords().second - x) + abs(item->first->getCoords().first - y);
67 enemy = item->first;
68 }
69 } catch (...) {
70 continue;
71 }
72 }
73 }
74 if (enemy == nullptr) return -4;
75 if (h > skillRange) return -5;
76
77 if (nameMorph == "undead" && dungeon.getHero().first->isOpenEnemy("livingEnemy")) {
78 dungeon.getListEnemies().push_back(
79 {new Undead(10, enemy->Damage(), enemy->getCoords(), "undead",
80 "livingEnemy",
81 enemy->getMaxHp(), enemy->getCurHp()),
82 &(dungeon.getMapLevel()[enemy->getCoords().second][enemy->getCoords().first])});
83 (--dungeon.getListEnemies().end())->first->setHeap(true);
84 (--dungeon.getListEnemies().end())->first->setFaction(enemy->getFaction());
85 enemy->die();
86 enemy->methodAfterDie(dungeon);
87 dungeon.getHero().first->getCharacteristics().mana(manaCost * skillLvl / 2);
88 } else if (nameMorph == "undeadSummoner" && dungeon.getHero().first->isOpenEnemy("livingSummoner")) {
89 dungeon.getListEnemies().push_back(
90 {new UndeadSummoner(10, enemy->Damage(), enemy->getCoords(), enemy->getMaxHp(), enemy->getCurHp()),
91 &(dungeon.getMapLevel()[enemy->getCoords().second][enemy->getCoords().first])});
