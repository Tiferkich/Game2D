11void LivingSummoner::doSomething(Dungeon &dungeon) {

16 Unit *someone = findTarget(dungeon);
17 if (someone == nullptr) {
18 status = Status::wandering;
19 srand(time(0));
20 int i = rand() % 4;
21 int dx[4] = {1, 0, -1, 0}; // смещения, соответствующие соседям ячейки
22 int dy[4] = {0, 1, 0, -1};
23 if (coords.first + dy[i] < 0 || coords.second + dx[i] < 0 ||
24 coords.second + dx[i] >= dungeon.getMapLevel().cols() ||
25 coords.first + dy[i] >= dungeon.getMapLevel().rows())
26 return;
27 try {
28 move(dungeon.getMapLevel(), {coords.first + dy[i], coords.second + dx[i]}, dungeon.getListEnemies());
29 } catch (...) {}
30 return;
31 }
32 try {
33 move(dungeon.getMapLevel(), {someone->getCoords().first, someone->getCoords().second},
34 dungeon.getListEnemies());
35 status = Status::aggresion;
36 } catch (...) {}
37 try {
38 summon(dungeon, {coords.first, coords.second}, dungeon.getHero().first->getLvl() / 2, curHP);
39 } catch (...) {}
40 try {
41 attack(*someone);
42 status = Status::aggresion;
43 } catch (...) {}
44}
