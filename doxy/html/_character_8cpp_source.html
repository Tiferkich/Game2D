67int Character::move(Move side, Dungeon &dungeon) {
68 sideLooking = side;
69 if (side == Move::down && dungeon.getMapLevel()[coords.first - 1][coords.second] != Ceil::wall &&
70 dungeon.getMapLevel()[coords.first - 1][coords.second] != Ceil::closeDoor) {
71 --coords.first;
72 } else if (side == Move::up && dungeon.getMapLevel()[coords.first + 1][coords.second] != Ceil::wall &&
73 dungeon.getMapLevel()[coords.first + 1][coords.second] != Ceil::closeDoor) {
74 ++coords.first;
75 } else if (side == Move::right && dungeon.getMapLevel()[coords.first][coords.second + 1] != Ceil::wall &&
76 dungeon.getMapLevel()[coords.first][coords.second + 1] != Ceil::closeDoor) {
77 ++coords.second;
78 } else if (side == Move::left && dungeon.getMapLevel()[coords.first][coords.second - 1] != Ceil::wall &&
79 dungeon.getMapLevel()[coords.first][coords.second - 1] != Ceil::closeDoor) {
80 --coords.second;
81 } else {
82 return -1;
83 }
84 dungeon.setHero({this, &dungeon.getMapLevel()[coords.first][coords.second]});
85 return 0;

89Ceil Character::close(Matrix<Ceil> &matrix) {
90 bool flag = false;
91 int dx[9] = {1, 0, -1, 0, 1, 0, 1, -1, -1}; // смещения, соответствующие соседям ячейки
92 int dy[9] = {0, 1, 0, -1, 1, 0, -1, -1, 1};
93 int x, y;
94 for (int i = 0; i < 6; i++) {
95 if (flag) { break; }
96 x = coords.second + dx[i];
97 y = coords.first + dy[i];
98 if (matrix[y][x] == Ceil::openDoor) flag = true;
99 }
100 if (!flag) throw std::invalid_argument("No open door");
101 matrix[y][x] = Ceil::closeDoor;
102 return Ceil::openDoor;
103}

173int Character::openEnemy(Dungeon &dungeon) {
174 if (countEssence == 0) return -3;
175 auto item = dungeon.getListEnemies().begin();
176 Enemy *enemy = nullptr;
177 for (item = dungeon.getListEnemies().begin(); item != dungeon.getListEnemies().end(); item++) {
178 if (item->first->getStatus() == Status::dead && item->first->getFaction() == Faction::livingEnemy &&
179 checkNear(item->first->getCoords()) && !isOpenEnemy(item->first->getName())) {
180 enemy = item->first;
181 }
182 }
183 if (enemy == nullptr) return -1;
184 if (enemy->openCost() > countEssence) return -2;
185 openEnemies.emplace_back(enemy->getName());
186 countEssence -= enemy->openCost();
187 return 0;
188}

206int Character::doSomething(int options[], Game &game) {
207 if (options == nullptr) return -1;
208 switch (options[0]) {
209 case 1:
210 move(Move::right, game.getAllMap()[getLvlPoint()]);
211 break;
212 case 2:
213 move(Move::left, game.getAllMap()[getLvlPoint()]);
214 break;
215 case 3:
216 move(Move::up, game.getAllMap()[getLvlPoint()]);
217 break;
218 case 4:
219 move(Move::down, game.getAllMap()[getLvlPoint()]);
220 break;
221 case 5:
222 try {
223 open(game.getAllMap()[getLvlPoint()].getMapLevel());
224 break;
225 } catch (...) {
226 break;
227 }
228 case 6:
229 try {
230 close(game.getAllMap()[getLvlPoint()].getMapLevel());
231 break;
232 } catch (...) {
233 break;
234 }
235 case 7:
236 takeEssense(game.getAllMap()[getLvlPoint()]);
237 break;
238 case 8:
239 if (uptoNewDungeon(game.getAllMap()[getLvlPoint()].getMapLevel()))
240 game.setCurLvl(game.getCurLvl() + 1);
241 break;
242 case 9:
243 if (game.getCurLvl() == 0) break;
244 if (uptoNewDungeon(game.getAllMap()[getLvlPoint()].getMapLevel()))
245 game.setCurLvl(game.getCurLvl() - 1);
246 break;
247 case 10: {
248 int forMorphism[] = {options[2]};
249 useSkill(Game::allSkills[options[1]], game.getAllMap()[getLvlPoint()], forMorphism);
250 break;
251 }
252 case 11:
253 upgradeSkill(Game::allSkills[options[1]]);
254 break;
255 case 12:
256 attackSomeone(game.getAllMap()[getLvlPoint()]);
257 break;
258 }
259 return 0;
260}
